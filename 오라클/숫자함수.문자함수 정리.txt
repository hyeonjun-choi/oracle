---------------------------------------------------------------------------------------------------------
	abs(n)
---------------------------------------------------------------------------------------------------------
		▶ 숫자 n의 절대값을 반환(음수를 양수로)
---------------------------------------------------------------------------------------------------------
	floor(n)
---------------------------------------------------------------------------------------------------------
		▶ 숫자 n이 정수면 그대로 반환하고, 실수면 실수보다 작으면서
			제일 가까운 정수를 반환. ( ex. -3.4  --> -4)
		▶ 좌표계에서 쿵 찍어서 왼쪽 정수값 (소수를 버려서 작아지는 것)
---------------------------------------------------------------------------------------------------------
	ceil(n)
---------------------------------------------------------------------------------------------------------
		▶ 숫자 n이 정수면 그대로 반환하고, 실수면 실수보다 크면서
			제일 가까운 정수를 반환.  ( ex. -3.4  --> -3)
		▶ 좌표계에서 쿵 찍어서 오른쪽 정수값
---------------------------------------------------------------------------------------------------------
	round(n,m)
---------------------------------------------------------------------------------------------------------
		▶ 숫자 n에서 소수 m+1 자리에서 반올림한 수 리턴 
		ex. round(3.4567,3)  ---> 3.457
---------------------------------------------------------------------------------------------------------
	trunc(n,m)
---------------------------------------------------------------------------------------------------------
		▶ 숫자 n에서 소수 m+1 자리부터 버림 내가 원하는 소수점 자리를 정해서 + 1
		ex. 
---------------------------------------------------------------------------------------------------------
	sign(n) ★★★★★★★★★★★★★★★★★★★★★★
---------------------------------------------------------------------------------------------------------
		▶ n>0 이면 1, n<0 이면 -1, n=0 이면 0을 반환
		
---------------------------------------------------------------------------------------------------------
	mod(n,m)
---------------------------------------------------------------------------------------------------------
		▶ n/m 후 나머지 값을 반환 mod(5,2) --- 특정 숫자를 골라 가려낼 때 사용
							ex. 3의 배수는 나머지 모두 0
---------------------------------------------------------------------------------------------------------
	power(n,m)
---------------------------------------------------------------------------------------------------------
		▶ n의 m승을 반환 --- power(5,2)

<60> 오라클에서 자주 사용되는 문자함수 예를 보여주세요.
■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■	

---------------------------------------------------------------------------------------------------------
	chr(n)
---------------------------------------------------------------------------------------------------------
		▶ 정수 n을 ascii(아스키) 문자기호로 리턴한다
		
---------------------------------------------------------------------------------------------------------
	uppper(s)
---------------------------------------------------------------------------------------------------------
		▶ 문자열 s를 대문자로 변환시켜 리턴한다 (가상으로 바꿔서)(영어 대상으로 포함, 한글과 동시에 있다면 써야함)

---------------------------------------------------------------------------------------------------------
	lower(s)
---------------------------------------------------------------------------------------------------------
		▶ 문자열 s를 소문자로 변환시켜 리턴한다 (가상으로 바꿔서)(영어 대상으로 포함, 한글과 동시에 있다면 써야함)

---------------------------------------------------------------------------------------------------------
	length(s)
---------------------------------------------------------------------------------------------------------
		▶ 문자열 s의 길이를 리턴한다 (문자의 길이 리턴)

---------------------------------------------------------------------------------------------------------
	substr(s, n1, n2)
---------------------------------------------------------------------------------------------------------
		▶ 문자열 s에서 n1번째 자리부터 n2개 까지의 문자를 리턴한다 (중간의 공백도 순서에 포함됨)
		▶ n2 생략 시 마지막까지 문자를 리턴한다
		▶ n1이 0일 경우, 1로 바꿔 실행된다
		▶ 자리수는 n>0 경우 왼쪽부터, n1<0 경우 오른쪽부터 셈한다 <중요>
		-----------------------------------
		select substr( 'You are not alone', 9, 3 ) from dual;
		select substr( 'You are not alone', 5) from dual;
		-----------------------------------
		덜 중요
		select substr( 'You are not alone', 0, 5 ) from dual;
		select substr( 'You are not alone', -5, 2 ) from dual;
		select substr( 'You are not alone', 5, 0 ) from dual; -> null값 나옴
---------------------------------------------------------------------------------------------------------
	trim(s)
---------------------------------------------------------------------------------------------------------
		▶ 문자열 s1의 맨 좌구측으 ㅣ공백을 제거하여 리턴한다
			제거 후 또 있으면 계속 제거 후 리턴한다

---------------------------------------------------------------------------------------------------------
replace( s1, s2, s3 ) ★★★★
---------------------------------------------------------------------------------------------------------
	▶문자열 s1에서 s2를 s3로 수정해서 리턴한다.
	▶<예> select replace(addr, '서울특별시','서울시')  from employee
	-----------------------------------------------------
	select replace( 'You are not alone', 'You', 'We' ) from dual; (임시로 you를 we로 바꿔서)
	select replace( 'You are not alone', 'not' ) from dual; (not을 없애라)
	select replace( 'You are not alone', 'not', '' ) from dual;
	-----------------------------------------------------
	select translate( 'You are not alone', 'You', 'We' ) from dual; (각 단어 하나하나씩 대응하면서 바꿔달라)(잘안씀)
---------------------------------------------------------------------------------------------------------
initcap( s )
---------------------------------------------------------------------------------------------------------
	▶문자열 s에서 각 단어의 첫 문자만 대문자로 변환하고 나머지는
	  모두 소문자로 리턴한다.
	▶select  initcap( 'abc def' ) from dual; --> Abc Def 로 바꾼다
---------------------------------------------------------------------------------------------------------
lpad( s1, n, s2 )
---------------------------------------------------------------------------------------------------------
	▶n 자리 만큼 (자리) 확보하고 s1을 오른쪽 정렬 후 왼쪽에 생긴 빈 공백에
	  문자열 s2를 채워 리턴한다.
	▶select lpad( 'abc', 7, ' ' ) from dual;
---------------------------------------------------------------------------------------------------------
rpad( s1, n, s2 )
---------------------------------------------------------------------------------------------------------
	▶n 자리 만큼 (자리) 확보하고 s1을 왼쪽 정렬 후 오른쪽에 생긴 빈 공백에
	  문자열 s2를 채워 리턴한다.
	▶select rpad( 'abc', 7, '#' ) from dual;
---------------------------------------------------------------------------------------------------------
instr( s1, s2, n1, n2 )
---------------------------------------------------------------------------------------------------------
	▶( s1:타겟문자, s2:찾을문자, 에서 n1:검색시작위치, n2:몇번째 )
	<주의>오라클에서 위치를 따질 때 시작은 0이 아니라 1이다.
	▶문자열 s1 에서 문자열 s2 가 있는 위치 반환. (자바의 indexof(~) 와 비슷한 기능)
	select
		instr( 'Every Sha-la-la-la Every wo-o-wo-o', 'la' ),
		instr( 'Every Sha-la-la-la Every wo-o-wo-o', 'la', 1),
		instr( 'Every Sha-la-la-la Every wo-o-wo-o', 'la', 1, 2),
		instr( 'Every Sha-la-la-la Every wo-o-wo-o', 'la', 12, 2)
	from dual;


	select * from employee where emp_name like '%김%'
	select * from employee where instr(emp_name, '김')>0;
	------------------------------------------------------------
	select * from employee where instr(emp_name, '김')=1;
	select * from employee where emp_name like '김%'				

<107>서브쿼리(subsqery)
	select, insulet, update, delete 구문 안에 들어 있는 또 다른 select 문을 말한다.
	from절에 나오는 select는 버스쿼리가 아니라 inline(인라인뷰) 라고 말함

서브쿼리 사용시 주의점
	select, update, delete 안의 서브쿼리는 꼭 () 로 묶는다
	경우에 따라 조인 대신 서브쿼리를 써도 같은 결과 낸다. 이때 join보다 서브쿼리의 부하가 더 많이 걸린다.
---------------------------------------------------------------------------------------------------------
서브쿼리 종류
---------------------------------------------------------------------------------------------------------
	비상관쿼리
		[서브 쿼리]와 [외부 쿼리]가 연관성이 없다.
		[서브 쿼리] 실행 후의 결과값을 가지고 [외부 쿼리] 실행된다.
		
		<예> 평균 연봉 이상 받는 직원을 검색
		select * from employee where salary >= (select avg(salary) from employee)
	상관쿼리
		[서브 쿼리]와 [외부 쿼리]가 연관성이 있다.	
		[서브 쿼리]와 [외부 쿼리]가 서로 통신하면서 쿼리가 진행된다.
